📋 How to Set Up (5 min)
What this does: Every time you add or edit an entry, the app can automatically send it to a Google Sheet you own. You need to set up a free Google Apps Script once.
1
Go to sheets.google.com and create a new spreadsheet. Name it "MoneyMOM".
2
Click Extensions → Apps Script in the menu bar.
3
Delete all existing code in the editor, then paste the script below.
4
Click Deploy → New Deployment. Choose type: Web App. Set "Who has access" to Anyone. Click Deploy.
5
Copy the Web App URL it gives you, paste it in the field above, and click Save URL.
6
Click Sync All Data to Sheet to push all 102 historical entries.
Paste this into Apps Script:
function doGet(e) { try { const sheet = SpreadsheetApp .getActiveSpreadsheet() .getActiveSheet(); const lastRow = sheet.getLastRow(); if (lastRow < 2) { return ContentService .createTextOutput(JSON.stringify({rows:[], debug:'sheet has no data rows'})) .setMimeType(ContentService.MimeType.JSON); } const data = sheet.getRange(2, 1, lastRow - 1, 6).getValues(); const rows = data .filter(r => r[0] !== '' && r[0] !== null) .map(r => { const d = r[1]; let dateStr; if (d && typeof d.getTime === 'function') { const y = d.getFullYear(); const m = String(d.getMonth() + 1).padStart(2, '0'); const day = String(d.getDate()).padStart(2, '0'); dateStr = y + '-' + m + '-' + day; } else { dateStr = String(d); } return { id: String(r[0]), date: dateStr, type: String(r[2]), amount: Number(r[3]), cat: String(r[4] || 'other'), comment: String(r[5] || '') }; }); return ContentService .createTextOutput(JSON.stringify({rows})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({rows:[], error: err.message})) .setMimeType(ContentService.MimeType.JSON); } } function doPost(e) { const sheet = SpreadsheetApp .getActiveSpreadsheet() .getActiveSheet(); // Add headers on first run if (sheet.getLastRow() === 0) { sheet.appendRow([ 'ID','Date','Type','Amount', 'Category','Note','Synced At' ]); sheet.getRange(1,1,1,7) .setFontWeight('bold') .setBackground('#1a3a6b') .setFontColor('#ffffff'); } const data = JSON.parse(e.postData.contents); // ── Handle DELETE ── if (data.deleteId) { const lastRow = sheet.getLastRow(); if (lastRow > 1) { const ids = sheet.getRange(2,1,lastRow-1,1) .getValues().flat(); const idx = ids.indexOf(data.deleteId); if (idx >= 0) { sheet.deleteRow(idx + 2); } } return ContentService .createTextOutput(JSON.stringify({ok:true,action:'deleted'})) .setMimeType(ContentService.MimeType.JSON); } // ── Handle ADD / EDIT ── const rows = data.rows || [data]; rows.forEach(row => { const lastRow = sheet.getLastRow(); const ids = lastRow > 1 ? sheet.getRange(2,1,lastRow-1,1).getValues().flat() : []; const idx = ids.indexOf(row.id); const newRow = [ row.id, row.date, row.type, row.amount, row.cat, row.comment, new Date().toLocaleString() ]; if (idx >= 0) { sheet.getRange(idx+2,1,1,7).setValues([newRow]); } else { sheet.appendRow(newRow); } }); return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); }
